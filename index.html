Java示例

import com.theokanning.openai.OpenAiApi;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.service.OpenAiService;

import java.util.ArrayList;
import java.util.List;

public class OpenAIExample {
    public static void main(String[] args) {
        // 设置API密钥和服务URL
        String token = "sk-xxxx";
        String baseUrl = "https://zzzzapi.com/v1";
        
        // 初始化服务
        OpenAiService service = new OpenAiService(token, baseUrl);
        
        // 创建消息列表
        List messages = new ArrayList<>();
        messages.add(new ChatMessage("user", "解释中转API的成本优势"));
        
        // 创建请求
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-3.5-turbo")
                .messages(messages)
                .build();
        
        // 发送请求并打印响应
        service.createChatCompletion(request).getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }
}
